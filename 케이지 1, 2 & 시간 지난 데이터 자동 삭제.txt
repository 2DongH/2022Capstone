CREATE database CAGE_1;

CREATE TABLE DHT11_1(
Seq INT AUTO_INCREMENT PRIMARY KEY,
Temperature INT,
Humidity INT,
Sensortime timestamp not null default current_timestamp 
);

CREATE TABLE Illuminance_1(
Seq INT AUTO_INCREMENT PRIMARY KEY,
Brightness INT,
Sensortime timestamp not null default current_timestamp 
);

CREATE TABLE Soil_Moisture_1(
Seq INT AUTO_INCREMENT PRIMARY KEY,
Moisture INT,
Sensortime timestamp not null default current_timestamp 
);
